    ```typescript
    export interface SearchResult {
      id: number
      title: string
      slug: string
      excerpt?: string
      type: 'blog_post' | 'case_study' | 'guide'
    }

    export async function fetchSearchResults(query: string): Promise<SearchResult[]> {
      if (query.length < 2) return []
      return fetchFromApi<SearchResult[]>(`/api/search?q=${encodeURIComponent(query)}`)
        .catch(() => [])
    }
    ```

    **2. Create `src/app/search/page.tsx`** (Server Component — reads searchParams from URL):
    ```typescript
    import { fetchSearchResults } from '@/lib/api'
    import type { Metadata } from 'next'
    import { generateSeoMetadata } from '@/lib/seo'

    interface Props {
      searchParams: Promise<{ q?: string }>
    }

    export async function generateMetadata({ searchParams }: Props): Promise<Metadata> {
      const { q } = await searchParams
      return generateSeoMetadata('page', 'search', {
        title: q ? `Search: "${q}" — Buildera` : 'Search — Buildera',
        description: 'Search Buildera\'s blog posts, case studies, and guides.',
        path: '/search',
      })
    }

    export default async function SearchPage({ searchParams }: Props) {
      const { q = '' } = await searchParams
      const results = q.length >= 2 ? await fetchSearchResults(q) : []

      // Group results by type
      const blogs = results.filter((r) => r.type === 'blog_post')
      const caseStudies = results.filter((r) => r.type === 'case_study')
      const guides = results.filter((r) => r.type === 'guide')
      const services = results.filter((r) => r.type === 'service')

      return (
        <>
          {/* Search Hero */}
          <section className="py-16 bg-background border-b border-border">
            <div className="container mx-auto px-8 max-w-4xl">
              <h1 className="text-3xl font-bold mb-6">Search Buildera</h1>
              <form method="GET" action="/search" className="flex gap-3">
                <input
                  type="search"
                  name="q"
                  defaultValue={q}
                  placeholder="Search blog posts, case studies, guides..."
                  className="flex-1 px-4 py-3 border border-border rounded-xl text-sm focus:outline-none focus:ring-2 focus:ring-[var(--brand-primary)]"
                  minLength={2}
                />
                <button
                  type="submit"
                  className="px-6 py-3 bg-[var(--brand-primary)] text-white rounded-xl text-sm font-medium hover:opacity-90 transition-opacity"
                >
                  Search
                </button>
              </form>
              {q && (
                <p className="mt-4 text-sm text-muted-foreground">
                  {results.length === 0
                    ? `No results found for "${q}"`
                    : `${results.length} result${results.length !== 1 ? 's' : ''} for "${q}"`}
                </p>
              )}
            </div>
          </section>

          {/* Results */}
          <section className="py-16 bg-[var(--brand-surface)]">
            <div className="container mx-auto px-8 max-w-4xl">
              {results.length === 0 && q.length >= 2 && (
                <div className="text-center py-12 text-muted-foreground">
                  <p className="text-lg">No results for "{q}".</p>
                  <p className="mt-2 text-sm">Try a different search term, or <a href="/blog" className="text-[var(--brand-primary)] underline">browse the blog</a>.</p>
                </div>
              )}

              {/* Render each group with heading */}
              {[
                { label: 'Blog Posts', items: blogs, href: (s: SearchResult) => `/blog/${s.slug}` },
                { label: 'Case Studies', items: caseStudies, href: (s: SearchResult) => `/case-studies/${s.slug}` },
                { label: 'Guides', items: guides, href: (s: SearchResult) => `/guides/${s.slug}` },
                { label: 'Services', items: services, href: (s: SearchResult) => `/services/${s.slug}` },
              ].filter(group => group.items.length > 0).map(group => (
                <div key={group.label} className="mb-12">
                  <h2 className="text-lg font-semibold mb-4 text-foreground">{group.label}</h2>
                  <ul className="flex flex-col gap-4">
                    {group.items.map(item => (
                      <li key={item.id}>
                        <a href={group.href(item)} className="block p-5 bg-white rounded-xl border border-border hover:border-[var(--brand-primary)] hover:shadow-md transition-all">
                          <p className="font-medium text-foreground">{item.title}</p>
                          {item.excerpt && <p className="mt-1 text-sm text-muted-foreground line-clamp-2">{item.excerpt}</p>}
                        </a>
                      </li>
                    ))}
                  </ul>
                </div>
              ))}

              {!q && (
                <div className="text-center py-12 text-muted-foreground">
                  <p>Enter a search term above to find blog posts, case studies, and guides.</p>
                </div>
              )}
            </div>
          </section>
        </>
      )
    }
    ```
