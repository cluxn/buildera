    ```typescript
    import { MetadataRoute } from 'next'

    const SITE_URL = process.env.NEXT_PUBLIC_SITE_URL ?? 'https://buildera.co'

    // Import all static slug arrays from data files
    import { CATEGORY_SLUGS, getServicesByCategory } from '@/data/services/index'
    // Import solution slugs (check src/data/solutions/ for the slug list)
    // Import industry slugs (check src/data/industries/ for the slug list)

    // Build complete service slug list: /services/{category}/{slug}
    const SERVICE_URLS: string[] = CATEGORY_SLUGS.flatMap((cat) =>
      getServicesByCategory(cat).map((s) => `/services/${cat}/${s.slug}`)
    )

    // Static pages
    const STATIC_PATHS = [
      '/', '/services', '/solutions', '/industries', '/about', '/contact',
      '/faq', '/how-we-work', '/book-a-call', '/privacy', '/terms',
      '/blog', '/case-studies', '/guides',
      ...CATEGORY_SLUGS.map((c) => `/services/${c}`),
      ...SERVICE_URLS,
      // Solution + industry paths: add after reading slug data files
    ]

    export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
      // Static entries
      const staticEntries: MetadataRoute.Sitemap = STATIC_PATHS.map((path) => ({
        url: `${SITE_URL}${path}`,
        lastModified: new Date(),
        changeFrequency: path === '/' ? 'weekly' : 'monthly',
        priority: path === '/' ? 1.0 : path.startsWith('/services/') && path.split('/').length === 4 ? 0.9 : 0.8,
      }))

      // Dynamic blog posts
      let blogEntries: MetadataRoute.Sitemap = []
      try {
        const data = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/api/blog-posts?per_page=200`, {
          next: { tags: ['blog_posts'], revalidate: 3600 },
        }).then((r) => r.json()) as { data: Array<{ slug: string; updated_at?: string }> }
        blogEntries = (data.data ?? []).map((post) => ({
          url: `${SITE_URL}/blog/${post.slug}`,
          lastModified: post.updated_at ? new Date(post.updated_at) : new Date(),
          changeFrequency: 'weekly' as const,
          priority: 0.7,
        }))
      } catch { /* fail gracefully — sitemap still works */ }

      // Dynamic case studies
      let caseStudyEntries: MetadataRoute.Sitemap = []
      try {
        const data = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/api/case-studies?per_page=200`, {
          next: { tags: ['case_studies'], revalidate: 3600 },
        }).then((r) => r.json()) as Array<{ slug: string; updated_at?: string }>
        caseStudyEntries = data.map((cs) => ({
          url: `${SITE_URL}/case-studies/${cs.slug}`,
          lastModified: cs.updated_at ? new Date(cs.updated_at) : new Date(),
          changeFrequency: 'monthly' as const,
          priority: 0.7,
        }))
      } catch {}

      // Dynamic guides
      let guideEntries: MetadataRoute.Sitemap = []
      try {
        const data = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/api/guides?per_page=200`, {
          next: { tags: ['guides'], revalidate: 3600 },
        }).then((r) => r.json()) as Array<{ slug: string; updated_at?: string }>
        guideEntries = data.map((g) => ({
          url: `${SITE_URL}/guides/${g.slug}`,
          lastModified: g.updated_at ? new Date(g.updated_at) : new Date(),
          changeFrequency: 'monthly' as const,
          priority: 0.7,
        }))
      } catch {}

      return [...staticEntries, ...blogEntries, ...caseStudyEntries, ...guideEntries]
    }
    ```

    Key: read the actual data files to get solution and industry slugs rather than assuming filenames. Check `src/data/solutions/` and `src/data/industries/` directories and use their slug exports.

    Update `app/Filament/Resources/SeoMetaResource.php` — add a HeaderAction to the table() method that pings the Next.js revalidation endpoint to regenerate the sitemap.

    In the `table(Table $table): Table` method, add `->headerActions([...])`:

    ```php
    use Filament\Tables\Actions\Action as TableAction;
    use Illuminate\Support\Facades\Http;

    ->headerActions([
        TableAction::make('regenerate_sitemap')
            ->label('Regenerate Sitemap')
            ->icon('heroicon-o-arrow-path')
            ->color('gray')
            ->action(function () {
                $secret = config('services.nextjs.revalidate_secret', env('NEXTJS_REVALIDATE_SECRET', ''));
                $url = config('services.nextjs.revalidate_url', env('NEXTJS_REVALIDATE_URL', ''));

                if (empty($url) || empty($secret)) {
                    \Filament\Notifications\Notification::make()
                        ->title('Sitemap regeneration not configured.')
                        ->warning()->send();
                    return;
                }

                $payload = json_encode(['tag' => 'sitemap']);
                $signature = hash_hmac('sha256', $payload, $secret);

                Http::withHeaders([
                    'x-revalidate-signature' => $signature,
                    'Content-Type' => 'application/json',
                ])->post($url, json_decode($payload, true));

                \Filament\Notifications\Notification::make()
                    ->title('Sitemap regeneration triggered.')
                    ->success()->send();
            }),
    ])
    ```

    Note: The sitemap is generated by Next.js at runtime — the revalidation tag 'sitemap' isn't a standard tag but serves as a trigger. For a fully static sitemap, the alternative is to call `revalidatePath('/sitemap.xml')`. The plan uses the existing revalidation infrastructure pattern.

    **Actually, since Next.js App Router sitemap.ts uses `revalidate` on its fetches, the sitemap auto-refreshes. The admin action should instead clear the relevant ISR tags (`blog_posts`, `case_studies`, `guides`) to force fresh data on next sitemap generation.**

    Revise the action to dispatch revalidation for `blog_posts`, `case_studies`, `guides` tags (which are what sitemap.ts fetches) — OR trigger a page revalidation for `/sitemap.xml`. Use whichever approach the existing RevalidationJob supports:

    ```php
    // Dispatch revalidation for all content tags to refresh sitemap data
    foreach (['blog_posts', 'case_studies', 'guides'] as $tag) {
        dispatch(new \App\Jobs\RevalidationJob($tag));
    }
    ```
