```php
class SitemapManagementPage extends Page
{
    protected static string $view = 'filament.pages.sitemap-management';
    protected static ?string $navigationGroup = 'SEO & Analytics';
    protected static ?string $navigationLabel = 'Sitemap';
    protected static ?string $navigationIcon = 'heroicon-o-map';
    protected static ?int $navigationSort = 5;

    public string $lastGeneratedAt = 'Never';
    public string $sitemapUrl = '';

    public function mount(): void
    {
        $this->lastGeneratedAt = Setting::get('sitemap_last_generated', 'Never');
        $this->sitemapUrl = config('app.url') . '/sitemap.xml';
    }

    public function regenerate(): void
    {
        // Regenerate sitemap using spatie/laravel-sitemap
        // The sitemap itself is generated by Next.js (src/app/sitemap.ts)
        // This action triggers ISR revalidation of the sitemap tag
        try {
            // Trigger revalidation of sitemap tag in Next.js
            $revalidateUrl = config('services.nextjs.revalidate_url', env('NEXTJS_REVALIDATE_URL'));
            $secret = config('services.nextjs.revalidate_secret', env('NEXTJS_REVALIDATE_SECRET'));
            
            if ($revalidateUrl && $secret) {
                $hmac = hash_hmac('sha256', 'sitemap', $secret);
                Http::withHeaders(['X-Revalidate-Secret' => $hmac])
                    ->post($revalidateUrl, ['tag' => 'sitemap']);
            }
            
            $timestamp = now()->toDateTimeString();
            Setting::set('sitemap_last_generated', $timestamp);
            $this->lastGeneratedAt = $timestamp;
            
            Notification::make()->title('Sitemap revalidation triggered')->success()->send();
        } catch (\Throwable $e) {
            Notification::make()->title('Revalidation failed: ' . $e->getMessage())->danger()->send();
        }
    }
}
```

Create Blade view at `resources/views/filament/pages/sitemap-management.blade.php`:
