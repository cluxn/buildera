    **`index()` method:**
    ```php
    public function index(Request $request): JsonResponse
    {
        $query = BlogPost::with('author')->scopePublished()->latest('published_at');

        if ($request->has('category') && $request->category !== '') {
            $query->where('category', $request->category);
        }
        if ($request->has('featured')) {
            $query->where('is_featured', true);
        }

        $posts = $query->paginate(12)->through(fn($post) => [
            'id' => $post->id,
            'title' => $post->title,
            'slug' => $post->slug,
            'excerpt' => $post->excerpt,
            'category' => $post->category,
            'tags' => $post->tags,
            'image_path' => $post->image_path ? Storage::url($post->image_path) : null,
            'image_alt' => $post->image_alt,
            'reading_time' => $post->reading_time,
            'published_at' => $post->published_at?->toISOString(),
            'author' => $post->author ? ['name' => $post->author->name, 'avatar' => $post->author->avatar ? Storage::url($post->author->avatar) : null] : null,
        ]);

        return response()->json($posts);
    }
    ```
